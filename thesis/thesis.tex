\chapter{ZI-Round}

\SetKwRepeat{Do}{do}{while}
\begin{algorithm}
	\KwIn{$x$: primal feasible solution of continuous relaxation.}
	\KwOut{rounded solution $x$ on success, none otherwise.}
	
	\Function{\upshape \texttt{ziround}}{
		\BlankLine
		$\Delta^{up}$ $\gets$ $\star$ maximum variable up-shifts $\star$ \;
		$\Delta^{down}$ $\gets$ $\star$ maximum variable down-shifts $\star$ \;
		$\mathcal{S}$ $\gets$ $\star$ constraint slacks $\star$ \;
		$\mathcal{SS}$ $\gets$ $\star$ singleton slacks $\star$ \;
		$\phi$ $\gets$ $\star$ objective function $\star$ \;
		$\phi(x)$ $\gets$ $\star$ objective value $\star$ \; 
		$ZI$ $\gets$ $\star$ fractionality function $\star$ \;
		$ZI(x)$ $\gets$ $\star$ solution fractionality $\star$ \;
		\BlankLine
		\Do{$\star$ variable roundings found $\star$}{
			\BlankLine
			\ForEach{$\star$ integer/binary variable $x_j$ $\star$}{
				\BlankLine
				\uIf{$\star$ $x_j$ non-fractional $\star$} {
					\BlankLine
					$\star$ compute new maximum up/down-shift of $x_j$ (threshold $\varepsilon = 1$) $\star$ \;
					\BlankLine
					\If{$(c_j \geq 0 \wedge \Delta_{j}^{down} = 1) \vee (c_j \leq 0 \wedge \Delta_{j}^{up} = 1)$}{
						\BlankLine
						$\star$ round $x_j$ to improve objective value $\phi(x)$ $\star$ \;
						$\star$ update slacks $\mathcal{S}$ and $\mathcal{SS}$ $\star$ \;
						$\star$ update objective value $\phi(x)$ $\star$ \;
					}
				}
				\ElseIf{$\star$ $x_j$ fractional $\star$}{
					\BlankLine
					$\star$ compute new maximum up/down-shift of $x_j$ (threshold $\varepsilon = 0.00001$) $\star$ \;
					\BlankLine
					\uIf{$\star$ both shifts improve fractionality $ZI(x)$ of the same amount $\star$}{
						\BlankLine
						$\star$ round $x_j$ to improve objective value $\phi(x)$ $\star$ \;
						$\star$ update slacks $\mathcal{S}$ and $\mathcal{SS}$ $\star$ \;
						$\star$ update objective value $\phi(x)$ $\star$ \;
						\BlankLine
					}
					\uElseIf{$\star$ rounding $x_j$ up improves fractionality $ZI(x)$ more $\star$}{
						\BlankLine
						$x_j$ $\gets$ $x_j + \Delta_{j}^{up}$ \;
						$\star$ update slacks $\mathcal{S}$ and $\mathcal{SS}$ $\star$ \;
						$\star$ update objective value $\phi(x)$ $\star$ \;
						\BlankLine
					}
					\ElseIf{$\star$ rounding $x_j$ down improves fractionality $ZI(x)$ more $\star$}{
						\BlankLine
						$x_j$ $\gets$ $x_j - \Delta_{j}^{down}$ \;
						$\star$ update slacks $\mathcal{S}$ and $\mathcal{SS}$ $\star$ \;
						$\star$ update objective value $\phi(x)$ $\star$ \;
						\BlankLine
					}
				}
				\BlankLine
			}
			\BlankLine
		}
		\BlankLine
	}
	\caption{ZI-Round.}
	\label{alg:ziround}
\end{algorithm}
